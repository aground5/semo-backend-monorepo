To access the service from outside the cluster:
{{- if contains "ClusterIP" .Values.service.type }}
  kubectl port-forward -n {{ .Values.namespace }} svc/{{ include "semo-backend-server.fullname" . }} {{ .Values.service.httpPort }}:{{ .Values.service.httpPort }}
  Then access the service at http://localhost:{{ .Values.service.httpPort }}
{{- else if contains "NodePort" .Values.service.type }}
  Get the NodePort:
  kubectl get -n {{ .Values.namespace }} svc/{{ include "semo-backend-server.fullname" . }} -o jsonpath='{.spec.ports[0].nodePort}'
  Then access using the NodePort and any node IP.
{{- else if contains "LoadBalancer" .Values.service.type }}
  Get the External IP:
  kubectl get -n {{ .Values.namespace }} svc/{{ include "semo-backend-server.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}'
  Then access the service using the External IP and port {{ .Values.service.httpPort }}.
{{- end }}
